Problem Statement

A program is required to read from the screen the length and width of a rectangular yard, and the length and width of the rectangular house that has been built on the yard. The algorithm should then compute the mowing time required to cut the grass around the house, at the rate of two square yards per minute. The program should display the length and width of the yard, the area of the yard, the length and width of hte house, the area of the house, the mowing area, and the mowing time.

// Calculate Mowing Time

// Declare variables
var houseArea = 0;
var houseLength = 0;
var houseWidth = 0;
var mowingArea = 0;
var mowingTime = 0;
var output = "";
var yardArea = 0;
var yardLength = 0;
var yardWidth = 0;

// inputs

// Prompt user for yard dimensions
yardLength = prompt("Enter the length of the yard");
yardWidth = prompt("Enter the width of the yard");

// Prompt user for house dimensions
houseLength = prompt("Enter the length of the house");
houseWidth = prompt("Enter the width of the house");

// processing

// Convert yard data to numbers
yardLength = Number(yardLength);
yardWidth = Number(yardWidth);

// Convert house data to numbers
houseLength = Number(houseLength);
houseWidth = Number(houseWidth);

// Calculate yard area
yardArea = yardLength * yardWidth;

// Calculate house area
houseArea = houseLength * houseWidth;

// Calculate mowing area
mowingArea = yardArea - houseArea;

// Calculate mowing time
mowingTime = mowingArea/2.0;

// output

// Display the data and calculations
output = output + "Your yard is " + yardLength + " yards by " +
                  yardWidth + " yards.\n";

output = output + "Your yard area is " + yardArea + " square yards.\n\n";


output = output + "Your house is " + houseLength + " yards by " +
                  houseWidth + " yards.\n";

output = output + "Your house area is " + houseArea + " square yards.\n\n";

output = output + "The mowing area is " + mowingArea + "\n";

output = output + "The calculated mowing time is " + mowingTime;

document.write(output);
