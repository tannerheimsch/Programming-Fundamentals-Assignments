// YOUR SOLUTION ALGORITHM GOES AFTER THIS LINE:

KidsVersusBedroomsAnalyzer
  Initialize numberOfKids to 0
  Initialize numberOfBedrooms to 0
  Initialize pluralKids to ""
  Initialize pluralBedrooms to ""
  Initialize output to ""

  Prompt user for numberOfKids with "Please enter how many kids were in your family when growing up:"
  Prompt user for numberOfBedrooms with "Please enter how many bedrooms you or your siblings had:"
  Convert numberOfKids to number
  Convert numberOfBedrooms to number

  IF numberOfKids = 1 THEN
    pluralKids = "kid"
  ELSE
    pluralKids = "kids"
  ENDIF
  IF numberOfBedrooms = 1 THEN
    pluralBedrooms = "bedroom"
    IF numberOfKids = 1 THEN
      messageEnd = "feel lucky!"
    ELSE IF numberOfKids = 2 THEN
      messageEnd = "feel cozy!"
    ELSE IF numberOfKids >= 3 THEN
      messageEnd = "feel crowded!"
  ENDIF
  ELSE IF numberOfBedrooms = 2 THEN
    pluralBedrooms = "bedrooms"
  IF numberOfKids = 1 THEN
      messageEnd = "feel spacious!"
  ELSE IF numberOfKids = 2 THEN
      messageEnd = "feel lucky!"
  ELSE IF numberOfKids >= 3 THEN
      messageEnd = "feel cozy!"
  ENDIF
  ELSE IF numberOfBedrooms >= 3 THEN
    pluralBedrooms = "bedrooms"
  IF numberOfKids = 1 THEN
      messageEnd = "have an empty feeling!"
  ELSE IF numberOfKids = 2 THEN
      messageEnd = "feel spacious!"
  ELSE IF numberOfKids >= 3 THEN
      messageEnd = "feel lucky!"
  ENDIF
END

  output = "With " + numberOfKids + " " + pluralKids + " and " + numberOfBedrooms + " "+ pluralBedrooms + " you must " + messageEnd;

  Display output
END
