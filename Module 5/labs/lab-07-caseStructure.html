SOLUTION ALGORITHM GOES HERE:

ASCIIValueChecker
  Initialize userLetter to ""
  Initialize userAsciiValue to ""
  Initialize userCharacterType to ""
  Initialize output to ""

  Prompt user for userLetter with "Please enter one of the following letters: A, B, C, D, or E."

  IF userLetter IS NOT a NUMBER THEN
    userCharacterType = "letter"
    CASE of userLetter TO upperCase
      "A": 
        userLetter = userLetter TO upperCase
        userAsciiValue = userLetter ASCII value
      "B":
        userLetter = userLetter TO upperCase
        userAsciiValue = userLetter ASCII value
      "C":
        userLetter = userLetter TO upperCase
        userAsciiValue = userLetter ASCII value
      "D":
        userLetter = userLetter TO upperCase
        userAsciiValue = userLetter ASCII value
      "E":
        userLetter = userLetter TO upperCase
        userAsciiValue = userLetter ASCII value
      default: 
        userAsciiValue = "an invalid input! Please try again."
    ENDCASE
  ELSE
    userCharacterType = "number"
    userAsciiValue = "an invalid input! Please try again"
  ENDIF
END

  output = "The " + userCharacterType + " " + userLetter + " is " + userAsciiValue + "."

  Display output
END
