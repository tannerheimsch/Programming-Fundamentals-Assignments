Example: Fahrenheit-Celsius Conversion

File: examples/example_Temperature.html

Problem Statement

A program is to be written that will accept 5 Fahrenheit temperatures, convert each one to Celsius, and display the converted temperatures on the page. After all 5 have been processed, the program will display the words, The program should only accept numerical input and should force the user to retry any values that are not numerical.

3. Solution Algorithm

The algorithm for this can be solved by breaking it apart into two components. First, a function that converts a temperature from Fahrenheit to Celsius. Then a main algorithm that asks for inputs in a loop.

While prompting for input, we check the current input to see if it's a number. If it is a number, we hand that number off to our conversion function, get the value back, and increment our counter by one.

If it's not a number, we state that it's not a number. We do not increment the counter. This makes the user enter the number again. The program will not proceed until they get it right.

When all the numbers are entered successfully, the output is displayed.

function ConvertFahrenheitToCelsius(fahrenheitTemperature) {
  var result = 0;
  fahrenheitTemperature = Number(fahrenheitTemperature);
  result = (fahrenheitTemperature - 32) * (5 / 9);
  return(result);
}

function ConvertFiveTemperatures() {

  // Declare variables
  var counter = 0;
  var currentFahrenheitTemperature;
  var currentCelsiusTemperature;
  var finalMessage = "All temperatures processed";
  var loopLimit = 5;
  var outputString = "";

  // Get user data, convert to Celsius and display
  while (counter < loopLimit) {
    currentFahrenheitTemperature = prompt("Enter Fahrenheit " +
      "temperature number " + (counter + 1) +
      " of " + loopLimit);
    if (isNAN(currentFahrenheitTemperature)) {
      alert("That's not a number. Please enter numbers only.");
    } else {
      currentCelsiusTemperature = ConvertFahrenheitToCelsius(currentFarenheightTemperature);
      outputString += currentFahrenheitTemperature + " degrees Fahrenheit " +
        "is " + currentCelsiusTemperature + " degrees Celsius.\n";
      counter++;
    }
  }

  // Output the final message
  document.write(outputString + finalMessage);
}
